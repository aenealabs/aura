== GETTING STARTED ==

{{- if .Values.frontend.ingress.enabled }}
1. Access the Aura UI:
   {{- range .Values.frontend.ingress.hosts }}
   https://{{ .host }}
   {{- end }}
{{- else }}
1. Access the Aura UI via port-forward:
   kubectl port-forward svc/{{ include "aura.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }} -n {{ .Release.Namespace }}
   Then open: https://localhost:{{ .Values.frontend.service.port }}
{{- end }}

{{- if .Values.api.ingress.enabled }}
2. API Endpoint:
   {{- range .Values.api.ingress.hosts }}
   https://{{ .host }}
   {{- end }}
{{- else }}
2. API Endpoint via port-forward:
   kubectl port-forward svc/{{ include "aura.fullname" . }}-api {{ .Values.api.service.port }}:{{ .Values.api.service.port }} -n {{ .Release.Namespace }}
{{- end }}
